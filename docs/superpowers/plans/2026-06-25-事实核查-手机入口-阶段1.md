> **For agentic workers:** REQUIRED SUB-SKILL: 用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 按任务逐个实现。步骤用 `- [ ]` 复选框追踪。
> **粒度说明**：本项目执行者熟悉代码、且大量复用现有 `runner` / `intake-worker` 模式，故任务按「组件级」拆分（每个任务含关键代码、测试要点、commit），样板代码指明模式来源而非全抄。每个任务做完跑 `bun test` 全绿再 commit。

**Goal:** 让事实核查能在手机上用——手机录入文本 / 链接，后台 `claude -p` 自动跑 `/factcheck`，结果同步回手机 Obsidian。（截图留阶段 2。）

**Architecture:** 复用现有 `runner` 架构。新增「私密核查页（凭密钥）→ Worker 私密路由（任务进 Cloudflare KV，不进公开 Issue）→ 独立核查 runner（Mac mini 轮询 KV、`claude -p` 跑 `/factcheck`、产出进 Obsidian）→ Obsidian Sync 回手机」。与公开站物理隔离。

**Tech Stack:** Cloudflare Worker + KV、Bun、原生 JS 前端、`claude -p` headless、launchd、nodemailer。

- [ ] **Step 1: 写 check.js 的三个处理函数**

复用现有 `safe-equal.js`（`safeEqual` 定长比较）、`ratelimit.js`（失败限流）模式。

```js
// services/intake-worker/src/check.js
import { safeEqual } from "./safe-equal.js";

// POST /check  —— 作者凭 CHECK_KEY 提交一条核查任务
export async function handleCheckSubmit(request, env, { now }) {
  const key = request.headers.get("x-check-key") || "";
  if (!env.CHECK_KEY || !safeEqual(key, env.CHECK_KEY)) {
    return json({ ok: false, error: "unauthorized" }, 401);
  }
  let body;
  try { body = await request.json(); } catch { return json({ ok: false, error: "bad json" }, 400); }
  const text = (body.text || "").trim();
  const link = (body.link || "").trim();
  if (!text && !link) return json({ ok: false, error: "empty" }, 400);
  if (text.length > 4000 || link.length > 1000) return json({ ok: false, error: "too long" }, 400);

  const id = crypto.randomUUID();
  const task = { text, link, status: "pending", createdAt: now() };
  await env.INTAKE_KV.put(`check:${id}`, JSON.stringify(task), { expirationTtl: 7 * 24 * 3600 });
  return json({ ok: true, id });
}

// GET /check/pending —— runner 凭 CHECK_RUNNER_SECRET 取待处理任务
export async function handleCheckPending(request, env) {
  if (!runnerAuthed(request, env)) return json({ ok: false }, 401);
  const list = await env.INTAKE_KV.list({ prefix: "check:" });
  const tasks = [];
  for (const k of list.keys) {
    const raw = await env.INTAKE_KV.get(k.name);
    if (!raw) continue;
    const t = JSON.parse(raw);
    if (t.status === "pending") tasks.push({ id: k.name.slice("check:".length), ...t });
  }
  return json({ ok: true, tasks });
}

// POST /check/<id>/done —— runner 标记完成
export async function handleCheckDone(request, env, id) {
  if (!runnerAuthed(request, env)) return json({ ok: false }, 401);
  const raw = await env.INTAKE_KV.get(`check:${id}`);
  if (!raw) return json({ ok: false, error: "not found" }, 404);
  const t = JSON.parse(raw);
  t.status = "done";
  await env.INTAKE_KV.put(`check:${id}`, JSON.stringify(t), { expirationTtl: 7 * 24 * 3600 });
  return json({ ok: true });
}

function runnerAuthed(request, env) {
  const s = request.headers.get("x-check-runner-secret") || "";
  return env.CHECK_RUNNER_SECRET && safeEqual(s, env.CHECK_RUNNER_SECRET);
}
function json(obj, status = 200) {
  return new Response(JSON.stringify(obj), { status, headers: { "content-type": "application/json" } });
}
```

```js
// poll.js —— 解析 /check/pending 响应（注入 fetch）
export async function fetchPendingChecks({ workerUrl, secret }, fetchImpl = fetch) {
  const r = await fetchImpl(`${workerUrl}/check/pending`, { headers: { "x-check-runner-secret": secret } });
  if (!r.ok) throw new Error(`pending ${r.status}`);
  const { tasks } = await r.json();
  return tasks;
}
export async function markCheckDone({ workerUrl, secret, id }, fetchImpl = fetch) {
  const r = await fetchImpl(`${workerUrl}/check/${id}/done`, { method: "POST", headers: { "x-check-runner-secret": secret } });
  if (!r.ok) throw new Error(`done ${r.status}`);
}
```

- [ ] **Step 2: runOnce 编排（runner.js，副作用注入）**

```js
// runner.js
export async function runOnce(config, deps) {
  const { fetchPending, markDone, runFactcheck, notify, log } = deps;
  const tasks = await fetchPending();
  let done = 0, fail = 0;
  for (const t of tasks) {
    const prompt = deps.buildPrompt(t);
    const code = await runFactcheck(prompt);      // spawn claude -p，返回退出码
    if (code !== 0) { fail++; log(`核查失败 ${t.id}（退出码 ${code}），留待重跑`); continue; }
    await markDone(t.id);
    done++;
    if (notify) await notify(t).catch(() => {});  // 发信尽力而为
  }
  return { processed: tasks.length, done, fail };
}
```
测试（全注入假实现）：多任务全成功 → done=N；某条 claude 退出码≠0 → 不 markDone、计入 fail、继续后续；notify 抛错不影响主流程。
